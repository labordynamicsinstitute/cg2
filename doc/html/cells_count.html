Sample SAS Code to Calculate the Number of Cells

libname dot '.';

proc sort data=dot.in(keep=personid firmid);
   by personid firmid;
run;

data out;
   set dot.in;
      by personid firmid;

   if first.firmid then output;
run;
/* The number of observations in the proc contents */
/* is the number of cells in your data             */

proc contents data=out;
run;
